@inject IDialogService DialogService

<FluentButton @onclick="@OpenDialog" Appearance="Appearance.Accent">
    Open Dialog
</FluentButton>

@code
{
    private void OpenDialog(MouseEventArgs e)
    {
        DemoLogger.WriteLine($"Open LaunchScreen for 4 seconds");

        SplashScreenParameters parameters = new()
        {
            Title = "Product name",
            SubTitle = "Suite name",
            LoadingText = "Loading...",
            Message = (MarkupString)"some <i>extra</i> text <strong>here</strong>",
            Width = "640px",
            Height = "480px",
            //OnDialogResult = DialogService.CreateDialogCallback(this, HandleSplash)
        };
        
        DialogService.ShowSplashScreen<SplashScreen>(this, HandleSplash, parameters);

    }

    private async Task HandleSplash(DialogResult result)
    {
        
        await Task.Run(() => DemoLogger.WriteLine($"Splash closed"));
    }
}
