@using System.Text.RegularExpressions
@{
    Regex pattern = new Regex("[;,<>&(){}!$^#@]");
    string ariaid = pattern.Replace(Title.Replace(' ', '-').ToLower(), "");
}
@if (!string.IsNullOrEmpty(Title)) 
{
    <h3 id="@ariaid">@Title</h3>
}

@if (Description != null) 
{
    <div class="demo-section-description">
        @Description
    </div>
}

<div class="demo-section-content">

    <FluentTabs ActiveId="@(!HideExample ? $"tab-example-{ariaid}" : $"tab-razor-{ariaid}")">
        @if (!HideExample) 
        {
            <FluentTab id="@("tab-example-"+@ariaid)" Label="Example" />
            <FluentTabPanel>
                <div class="demo-section-example">
                    @ChildContent
                    @if (Component is not null)
                    {
                        <DynamicComponent Type="Component" Parameters="ComponentParameters" />
                    }
                </div>
            </FluentTabPanel>
        }
        
        @if (!HideAllButExample && _hasCode)
        {
            @foreach (KeyValuePair<string, string> pair in _tabPanelsContent) 
            {
                string tab = Path.GetExtension(pair.Key);

                if (pair.Key.StartsWith(Component.Name))
                {
                    <FluentTab id="@("tab-"+@GetDisplayName(tab).Replace("#","").ToLower()+"-"+@ariaid)" Label="@GetDisplayName(tab)" />
                }
                else
                {
                    <FluentTab id="@("tab-"+tab.Replace("#","").ToLower()+"-"+@ariaid)" Label="@tab" />
                }
                <FluentTabPanel>

                    <CodeSnippet Language="@TabLanguageClass(tab)">@pair.Value</CodeSnippet>
                </FluentTabPanel>  
            }
        }
    </FluentTabs>
</div>
@if (ShowDownloads) 
{
    <div class="demo-section-downloads">
        <FluentSpacer />
        Download:&nbsp;
        @foreach (KeyValuePair<string, string> pair in _tabPanelsContent)
        {
            string source = pair.Key;
            <FluentAnchor Style="width: 85px;" Download="@source" Href="@($"_content/FluentUI.Demo.Shared/sources/{source}.txt")" Appearance="Appearance.Neutral">
                @if (source.StartsWith(Component.Name)) 
                {
                    @GetDisplayName(source.Substring(source.LastIndexOf('.')))
                }
                else 
                {
                    @source
                }
            </FluentAnchor>

            @if (source != _tabPanelsContent.Last().Key)
            {
                <span>&nbsp;</span>
            }
        }
    </div>
}
